// Create PRODUCT

create table PRODUCTS(
id INTEGER AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(60) NOT NULL,
brand VARCHAR(60) NOT NULL,
price DOUBLE NOT NULL,
)

// Create CART
create table CART(
id INTEGER AUTO_INCREMENT PRIMARY KEY
)



//Crete CARTSPRODUCTS

create table CARTSPRODUCTS (
id_product INT(6) UNSIGNED,
id_cart INT(6) UNSIGNED,
PRIMARY KEY(id_product, id_cart),
CONSTRAINT FK_ID_PRODUCT foreign key (id_product) references PRODUCT (id),
CONSTRAINT FK_ID_CART foreign key (id_cart) references CART (id) 
)
