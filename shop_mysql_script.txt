
create table CART(
ID INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY, 
IS_ACTIVE SMALLINT(1) 
);

create table PRODUCT(
ID INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
NAME VARCHAR(60) NOT NULL,
BRAND VARCHAR(60) NOT NULL,
PRICE DOUBLE NOT NULL,
IMAGE_URL VARCHAR(60) NOT NULL,
DESCRIPTION VARCHAR(600)
);

create table CARTSPRODUCTS (
ID_PRODUCT INT(6) UNSIGNED,
ID_CART INT(6) UNSIGNED,
AMOUNT INT(6) UNSIGNED,
PRIMARY KEY(ID_PRODUCT, ID_CART),
CONSTRAINT FK_ID_PRODUCT foreign key (ID_PRODUCT) references PRODUCT (ID),
CONSTRAINT FK_ID_CART foreign key (ID_CART) references CART (ID) 
);

create table ORDERS(
ID INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
EMAIL VARCHAR(60) NOT NULL,	
TYP_DOSTAWY VARCHAR(60),
TYP_ODBIORCY VARCHAR(60),
NAZWA VARCHAR(60) NOT NULL,
ADRES VARCHAR(60) NOT NULL,
CITY VARCHAR(60) NOT NULL,
POSTAL_CODE VARCHAR(60) NOT NULL,
ID_CART INT(6) UNSIGNED,
CONSTRAINT FK_CART_ORDER foreign key (ID_CART) references CART (id)
);
